CMD + C               # copy
CMD + V               # paste
CMD + Z               # undo
CMD + SHIFT + Z       # redo
CMD + R               # refresh
CMD + SHIFT + R       # hard refresh
CMD + T               # new tab in browser, terminal, etc.
